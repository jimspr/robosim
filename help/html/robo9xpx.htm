(set-window-menu menu)

Sets the menu of the program to the menu specified by menu. The format of a menu is a list of strings and actions. A list within a list corresponds to a submenu. The easiest way to understand this format is to look at an example.

(set-window-menu
    '(
        ("&File"
            "&New" (clear-simulation)
            "&Open..." (open-file)
            nil
            "&Print..." (printer-dialog)
            nil
            "E&xit" (exit)
        )
        ("&View"
            "&Copy" (copy-window)
            nil
            "&New View" (new-view)
            "New &3D View" (new-view3d 15)
            "&Set View..." (set-view-dialog)
        )
        ("&Options"
            "&Font..." (set-font-dialog)
            "&Lines..." (set-lines)
            "Double Buffer O&n"
                (set-double-buffer t)
            "Double Buffer O&ff"
                (set-double-buffer nil)
            "&Show Frames" (show-frames t)
            "&Don't Show Frames" (show-frames nil)
        )
        ("&Simulation"
            ("&Agent"
                "&T3" (setq robot t3-1)
                "&Puma" (setq robot puma-1)
                "&SS" (setq robot ss-1)
            )
            ("&Object"
                "&Box" (setq object box)
                "&Cyl" (setq object cyl)
            )
            "&Move Object" (simulate robot object)
            "Move &Robot" (move-it robot)
            "&Demo" (demo)
        )
        ("&Window"
            "&Cascade" (cascade-windows)
            ("&Tile"
                "Tile &Horizontal"
                    (tile-windows-h)
                "Tile &Vertical"
                    (tile-windows-v)
            )
            "&Arrange Icons" (arrange-icons)
        )
        ("&Help"
            "&About..." (about-dialog)
        )
    )
)
